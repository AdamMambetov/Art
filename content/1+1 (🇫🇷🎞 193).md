## Навигация

```dataviewjs
const path = dv.current().file.path
let artPath = '"Text/Art/content"'
if (!path.includes("Art"))
	artPath = '"content"'
const pages = dv
	.pages(artPath)
	.sort(p => p.created, "asc")
const regex = /.* (\d*)/g
const name = dv.current().file.name
let idx = Number([...name.matchAll(regex)][0][1]) - 1

let next = pages[idx + 1].file.name
let prev = pages[idx - 1].file.name

dv.paragraph(`[[${next} | Next Art: ${next}]]`)
dv.paragraph(`[[${prev} | Prev Art: ${prev}]]`)
```
